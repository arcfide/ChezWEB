\font\titlefont = cmr17

\newcount\sectc

\def\sect{%
  \advance\sectc by 1
  {\noindent\bf\the\sectc . }}
\def\endsect{\medskip}

\def\nsect#1#2{%
  \advance\sectc by 1
  {\noindent\bf\the\sectc . #2.} }
\def\endnsect{\medskip}

\def\chunk#1#2#3#4{%
  \bgroup
  %def\chunkimports{#2}
  %def\chunkexports{#3}
  %def\chunkcaptures{#4}
  \hfil\break$\langle$ #1 $\rangle \equiv$\hfil\break
  \parindent = 12pt
  \catcode`\#12
  \catcode`\_12
  \tt
  \obeylines\obeyspaces}
\def\endchunk{\egroup}

\def\code{%
  \bgroup
  \hfil\break$\langle * \rangle +\equiv$\hfil\break
  \parindent = 12pt
  \obeylines\obeyspaces
  \catcode`\#12
  \catcode`\_12
  \tt}
\def\endcode{\egroup}

\def\x#1;{\char"#1 }

\def\chunkref#1{%
  $\langle$ #1 $\rangle$}
